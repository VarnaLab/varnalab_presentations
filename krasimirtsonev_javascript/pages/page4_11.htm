4.11. Factory Pattern

		function Class1() {
			this.init = function() {}
		}
		function Class2() {
			this.init = function() {}
		}
		Factory.get = function(type){
			if (type == 1) {
				return new Class1();
			} else {
				return new Class2();
			}
		}
		var _instance = Factory.get(1);
		_instance.init();
