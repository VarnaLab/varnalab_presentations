4.5. Revealing Module Pattern

Автор: Chris Heilmann
Проблеми които решава:
- необходимостта да се пише името на класа когато се достъпва public метод от друг public метод
- дефиниране на public методи/променливи в обект 
		var Module = (function() {
			var _index = 0;			
			return {
				getIndex: function() {
					return _index;
				},
				increment: function() {
					_index = Module.getIndex() + 1;
				}
			};		
		})();
	
		var Module = (function() {
			var _index = 0;	
			var getIndex = function() {
				return _index;
			};
			var increment = function() {
				_index = getIndex() + 1;
			};
			return {
				getIndex: getIndex,
				increment: increment
			};		
		})();
