4.4. Module Pattern

* Прилича на Singleton pattern

		var Module = (function() {
			var _index = 0;			
			return {
				increment: function() {
					_index += 1;
				},
				getIndex: function() {
					return _index;
				}
			};		
		})();
		// недостатък: нямаме достъп до private промеливи от методи дефинирани по късно.
