1.3. Вмъкване посредством ajax и eval()

		$(document).ready(function() {
			$.ajax({
				url: "./module.js",
				success: function(result){
					var module = eval(result);
					module.getTitle();
				}
			});
		});
		
		
		// module.js
		(function(){
			
			var title = "The module !!!";
			
			return {
				getTitle: function() {
					alert(title);
				}
			};
			
		})();
