4.7. Prototype Pattern

* Prototype Pattern === inheritance
* Prototype pattern е базиран на концепцията за наследяване чрез прототипи, където създаме обекти, които действат като прототип за други обекти.

Object.create(proto [, propertiesObject ])

		var vehicle = {
			getModel : function(){
				console.log('The model of this vehicle is..' + this.model);
		   }
		};
		var car = Object.create(vehicle, {
			'id' : {
				value: MY_GLOBAL.nextId(),
				enumerable: true /*writable:false, configurable:false by default*/
			},
			'model':{
				value: 'Ford',
				enumerable: true
			}
		});
	
		function extend(ChildClass, ParentClass) {
			ChildClass.prototype = new ParentClass();
			ChildClass.prototype.constructor = ChildClass;
		}
