		function extend(Child, Parent) {
			var F = function(){};
			F.prototype = Parent.prototype;
			Child.prototype = new F();
			Child.prototype.constructor = Child;
			Child.uber = Parent.prototype;
		}
	
		var Shape = function() {
			this.name = "shape";
		};
		Shape.prototype = {
			getName:function() {
				return this.name;
			}
		};
		var Rect = function () {
			this.name = "rect";
		};
		extend(Rect, Shape);
