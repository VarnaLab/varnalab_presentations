3.3. Класове (public/private методи и променливи)

Дефиниране

		var Products = function() {
			
		};
		var Cart = function(username, password) {
			
		};

Променливи

		var Products = function() {
			// private
			var _items = [];
			// public
			this.orderStatus = "paid";
		};
		Products.prototype = {
			orderStatus: "paid"
		};

Методи

		var Products = function() {
			// private
			function checkStatus() {
				...
			}
			// public
			this.getItems = function() {
				...
			}
		};
		Products.prototype = {
			getItems: function() {
				...
			}
		};

Инициализация

		var p = new Products();
		/*
			p.orderStatus
			p.getItems
		*/

Примери

		var Cart = function() {
			
			var _items = [];
			
			this.addItem = function(item) {
				_items.push(item);
			};
			this.isEmpty = function() {
				return _items.length == 0 ? true : false;
			};
		
		};
	
		var Cart = function() {
			
			var _items = [];
			
			function addItem(item) {
				_items.push(item);
			};
			function isEmpty() {
				return _items.length == 0 ? true : false;
			};
			
			return {
				addItem:addItem,
				isEmpty:isEmpty
			};
		
		};
	
		var Cart = function() {
			
			// var _items = [];
			this.items = [];
		
		};
		Cart.prototype = {
			addItem:function(item) {
				this.items.push(item);
			},
			isEmpty:function() {
				return items.length == 0 ? true : false;
			}
		}
