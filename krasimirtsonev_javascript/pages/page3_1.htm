3.1. Що е то обект (prototype, prototype chain)

Primitive Data Types

		var isProgrammer = false; // boolean
		var age = 25; // number
		var name = "John Black"; // string
		var job; // undefined
		var salary = null; // променливата е дефинирана но нейната стойност е "нищо"

Other types

		var devs = new Array(); // array
		var sales = [], designes = []; // array
		
		var team = new Object(); // object
		team.devs = devs;
		team.sales = sales;
		team.designes = designes;
		team.status = new Object();
		team.retire = function() { };
		var team = { devs:[], sales:[], designes:[] } // object
