4.13. Decorator Pattern

- добавяне на нова функционалност 
		var User = function() {
			this.type = "";
			this.deleted = false;
		};
		
		var Admins = function() {
		
			var user = new User();
			user.type = "admin";
			
			this.getType = function() {
				return user.type;
			};
			
		};
		
		var Author = function() {
		
			var user = new User();
			user.type = "author";
			
			this.getType = function() {
				return user.type;
			};
			
		};
